CREATE TABLE public.assessments (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL,
  subject text NOT NULL DEFAULT 'math'::text,
  score integer NOT NULL CHECK (score >= 0 AND score <= 100),
  total_questions integer NOT NULL DEFAULT 12,
  knowledge_map jsonb NOT NULL DEFAULT '{}'::jsonb,
  responses jsonb NOT NULL DEFAULT '[]'::jsonb,
  avg_response_time integer,
  avg_confidence_score integer,
  avg_pattern_recognition integer,
  avg_conceptual_understanding integer,
  status text DEFAULT 'completed'::text CHECK (status = ANY (ARRAY['in_progress'::text, 'completed'::text])),
  started_at timestamp with time zone,
  completed_at timestamp with time zone DEFAULT timezone('utc'::text, now()),
  created_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  CONSTRAINT assessments_pkey PRIMARY KEY (id),
  CONSTRAINT assessments_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
